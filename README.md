```bash
npm install --save react-verification-code
```

## Usage

```tsx
import React, { Component } from 'react'

import MyComponent from 'react-verification-code'
import 'react-verification-code/dist/index.css'

class Example extends Component {
  render() {
    return <MyComponent />
  }
}
```
